```python
def test_via_subgraph_tag_present() -> None:
    provider = ScriptedProvider([
        json.dumps({"action": "tool", "tool_name": "repeat_message", "args": {"message": "hello"}}),
        json.dumps({"action": "finish", "answer": "done"}),
    ])
    orch = LangGraphOrchestrator(provider=provider, max_steps=10)
    result = orch.run("Say hello")
    tool_entries = [e for e in result["state"]["tool_history"] if e.get("tool") != "finish"]
    assert any(e.get("via_subgraph") for e in tool_entries), "No via_subgraph=True in tool_history"
```

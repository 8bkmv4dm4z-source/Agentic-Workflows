```python
TaskComplexity = Literal["planning", "evaluation", "error_recovery", "tool_selection", "continuation"]

_STRONG_TASKS: frozenset[TaskComplexity] = frozenset({"planning", "evaluation", "error_recovery"})

class ModelRouter:
    def __init__(self, strong_provider: ChatProvider, fast_provider: ChatProvider | None = None) -> None:
        self._strong = strong_provider
        self._fast = fast_provider or strong_provider  # falls back to strong when None

    def route(self, task_complexity: TaskComplexity) -> ChatProvider:
        if task_complexity in _STRONG_TASKS:
            return self._strong
        return self._fast  # tool_selection, continuation → fast

    @property
    def has_dual_providers(self) -> bool:
        return self._strong is not self._fast
```

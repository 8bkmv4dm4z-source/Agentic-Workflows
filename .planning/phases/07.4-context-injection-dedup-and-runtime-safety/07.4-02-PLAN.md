From src/agentic_workflows/orchestration/langgraph/context_manager.py (current cascade call site, lines ~595-602):

```python
# CURRENT (no timeout guard):
if cache_key not in self._cascade_cache:
    self._cascade_cache[cache_key] = self._store.query_cascade(
        goal_text,
        embedding=embedding,
```

REPLACE the direct `self._store.query_cascade(...)` call with:

```python
from concurrent.futures import ThreadPoolExecutor

_CASCADE_TIMEOUT_SECONDS: float = 2.0   # module-level constant

# Inside the if cache_key not in self._cascade_cache: block:
try:
    with ThreadPoolExecutor(max_workers=1) as executor:
        future = executor.submit(
            self._store.query_cascade,
            goal_text,
            embedding=embedding,
            top_k=3,
        )
        hits_result = future.result(timeout=_CASCADE_TIMEOUT_SECONDS)
    self._cascade_cache[cache_key] = hits_result
except TimeoutError:
    self._logger.warning(
        "query_cascade timeout after %.1fs — falling back to []",
        _CASCADE_TIMEOUT_SECONDS,
    )
    self._cascade_cache[cache_key] = []
except Exception as exc:  # noqa: BLE001
    self._logger.debug("cross-run cascade query failed (non-fatal): %s", exc)
    self._cascade_cache[cache_key] = []
```

NOTE: The existing outer `except Exception` block in `build_planner_context_injection()` (line ~607) already catches any exceptions from this whole block. The inner try/except here is specifically to differentiate the timeout warning from other errors.

NOTE: In Python 3.12 (project runtime), `concurrent.futures.TimeoutError` IS `builtins.TimeoutError`. Catching plain `TimeoutError` is sufficient. Do NOT import `TimeoutError as FuturesTimeoutError` — that alias is not needed in 3.12 and adds noise.

    ```python
    import time
    from unittest.mock import MagicMock
    from agentic_workflows.orchestration.langgraph.context_manager import ContextManager

    class TestCascadeTimeout:
        def _make_state(self, goal: str = "Sort array", run_id: str = "run-1") -> dict:
            return {"missions": [goal], "mission_contexts": {}, "messages": [],
                    "run_id": run_id, "step": 1}

        def test_slow_cascade_completes_within_timeout(self):
            mock_store = MagicMock()
            def slow(*a, **kw):
                time.sleep(5)
                return []
            mock_store.query_cascade.side_effect = slow
            cm = ContextManager(mission_context_store=mock_store)
            state = self._make_state()
            start = time.monotonic()
            result = cm.build_planner_context_injection(state)
            elapsed = time.monotonic() - start
            assert isinstance(result, str)
            assert elapsed < 3.0, f"Took {elapsed:.1f}s — timeout not enforced"

        def test_slow_cascade_returns_empty_cross_run(self):
            mock_store = MagicMock()
            mock_store.query_cascade.side_effect = lambda *a, **kw: (__import__('time').sleep(5), [])[1]
            cm = ContextManager(mission_context_store=mock_store)
            result = cm.build_planner_context_injection(self._make_state())
            assert "[Cross-run] Similar:" not in result

        def test_fast_cascade_still_works(self):
            from agentic_workflows.storage.mission_context_store import MissionContextResult
            mock_store = MagicMock()
            mock_store.query_cascade.return_value = [
                MissionContextResult(id=1, run_id="r", mission_id="m", goal="g",
                                     summary="s", tools_used=[], score=0.9, source_layer="L0")
            ]
            cm = ContextManager(mission_context_store=mock_store)
            result = cm.build_planner_context_injection(self._make_state())
            assert "[Cross-run] Similar:" in result
    ```
